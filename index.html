Пример моего кода


    int bfs(
        std::vector& used,
        const std::map>& children,
        int point) {

        std::set points = children.at(point);
        for (auto const& i : points) {
            used[i] = true;
            std::set i_children = children.at(i);
            for (auto const& j : i_children) {
                if (!used[j]) {
                    points.insert(j);
                    used[j] = true;
                }
            }
        }

        return points.size();
    }
