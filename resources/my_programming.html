Пример моего кода


    int bfs(
        std::vector<bool>& used,
        const std::map<int, std::set<int>>& children,
        int point) {

        std::set<int> points = children.at(point);
        for (auto const& i : points) {
            used[i] = true;
            std::set<int> i_children = children.at(i);
            for (auto const& j : i_children) {
                if (!used[j]) {
                    points.insert(j);
                    used[j] = true;
                }
            }
        }

        return points.size();
    }
